Problema 2

Conjuntos difusos tipo-II

FIS de tipo-I

%Creamos el FIS tipo-I usando sugfis.
fis1 = sugfis;
%Se a√±aden las variables de entrada.
fis1 = addInput(fis1,[-1 1],'Name','E');
fis1 = addInput(fis1,[-1 1],'Name','delE');
%Se agregan tres MFs triangulares uniformemente distribuidas y sobrepuestas a cada entrada.
fis1 = addMF(fis1,'E','trimf',[-2 -1 0],'Name','N');
fis1 = addMF(fis1,'E','trimf',[-1 0 1],'Name','Z');
fis1 = addMF(fis1,'E','trimf',[0 1 2],'Name','P');
fis1 = addMF(fis1,'delE','trimf',[-2 -1 0],'Name','N');
fis1 = addMF(fis1,'delE','trimf',[-1 0 1],'Name','Z');
fis1 = addMF(fis1,'delE','trimf',[0 1 2],'Name','P');
%Se plotean las MFs de entrada.
figure
subplot(1,2,1)
plotmf(fis1,'input',1)
title('Input 1')
subplot(1,2,2)
plotmf(fis1,'input',2)
title('Input 2')
%Se a√±ade la variable de salida al FIS.
fis1 = addOutput(fis1,[-1 1],'Name','U');
%Se agregan funciones constantes distribuidas uniformemente a la salida.
fis1 = addMF(fis1,'U','constant',-1,'Name','NB');
fis1 = addMF(fis1,'U','constant',-0.5,'Name','NM');
fis1 = addMF(fis1,'U','constant',0,'Name','Z');
fis1 = addMF(fis1,'U','constant',0.5,'Name','PM');
fis1 = addMF(fis1,'U','constant',1,'Name','PB');
%Se a√±aden las reglas al FIS. Estas reglas crearan un superficie de control proporcional.
rules = [...
"E==N & delE==N => U=NB"; ...
"E==Z & delE==N => U=NM"; ...
"E==P & delE==N => U=Z"; ...
"E==N & delE==Z => U=NM"; ...
"E==Z & delE==Z => U=Z"; ...
"E==P & delE==Z => U=PM"; ...
"E==N & delE==P => U=Z"; ...
"E==Z & delE==P => U=PM"; ...
"E==P & delE==P => U=PB" ...
];
fis1 = addRule(fis1,rules);
%Ploteamos la superficie de control
figure
gensurf(fis1)
title('Superficie de control del FIS tipo-I')

FIS de tipo-II

%Convertimos el FIS tipo-I a un FIS tipo-II.
fis2 = convertToType2(fis1);
%Se define la zona de incertidumbre para las MFs de entrada.
scale = [0.2 0.9 0.2;0.3 0.9 0.3];
for i = 1:length(fis2.Inputs)
for j = 1:length(fis2.Inputs(i).MembershipFunctions)
fis2.Inputs(i).MembershipFunctions(j).LowerLag = 0;
fis2.Inputs(i).MembershipFunctions(j).LowerScale = scale(i,j);
end
end
%Ploteamos las MFs de entrada de tipo-II
figure
subplot(1,2,1)
plotmf(fis2,'input',1)
title('Input 1')
subplot(1,2,2)
plotmf(fis2,'input',2)
title('Input 2')
%La zona de incertidumbre produce una superficie de control no lineal
figure
gensurf(fis2)
title('Superficie de control de un FIS tipo-II')
